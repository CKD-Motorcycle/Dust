Stateful API

int packet_count(int seconds)

Takes the number of seconds elapsed since the last call and returns the number of packets to send

void put_encoded(int len, byte[] bytes)

Takes number of bytes and the actual bytes and adds them to the decoding queue

void put_decoded(int len, byte[] bytes)

Takes number of bytes and the actual bytes and adds them to the encoding queue

int encoded_ready()

Returns the number of bytes available in the encoded queue

int decoded_ready()

Returns the number of bytes available in the decoded queue

byte[] get_encoded(int len)

Takes number of bytes and returns actual bytes from the front of the encoded queue, removing the bytes from the queue

byte[] get_decoded(int len)

Takes number of bytes and returns actual bytes from the front of the decoded queue, removing the bytes from the queue

void loadKeys(int len, byte[] path)

Takes number of bytes and the actual bytes and loads keys from the specified path

void put_encrypted(int len, byte[] bytes)

Takes number of bytes and the actual bytes and adds them to the decoding queue

void put_decrypted(int len, byte[] bytes)

Takes number of bytes and the actual bytes and adds them to the encoding queue

int encrypted_ready()

Returns the number of bytes available in the encrypted queue

int decrypted_ready()

Returns the number of bytes available in the decrypted queue

byte[] get_encrypted(int len)

Takes number of bytes and returns actual bytes from the front of the encrypted queue, removing the bytes from the queue

byte[] get_decrypted(int len)

Takes number of bytes and returns actual bytes from the front of the decrypted queue, removing the bytes from the queue
